#include <iostream>  // Підключення бібліотеки для вводу та виводу даних

using namespace std;  // Використання стандартного простору імен, щоб не писати std:: перед стандартними функціями та об'єктами

int main() {  // Початок основної функції програми
  double x, b, c, y;  // Оголошення змінних типу double для зберігання значень x, b, c і y

  cout << "Enter the value of x: ";  // Виведення повідомлення користувачу з проханням ввести значення для x
  cin >> x;  // Зчитування значення x, введеного користувачем, і збереження його в змінну x

  cout << "Enter the value of b: ";  // Виведення повідомлення користувачу з проханням ввести значення для b
  cin >> b;  // Зчитування значення b, введеного користувачем, і збереження його в змінну b

  cout << "Enter the value of c: ";  // Виведення повідомлення користувачу з проханням ввести значення для c
  cin >> c;  // Зчитування значення c, введеного користувачем, і збереження його в змінну c

  if (x < 2) {  // Перевірка умови: якщо значення x менше 2
    y = (x - 9 * b * x) / (x - 2 * b * x * x);  // Обчислення значення y за формулою, якщо умова виконується
  } else {  // Якщо значення x не менше 2 (тобто x >= 2)
    y = (x - 2 * c * x) / (x - 5 * c * x * x * x);  // Обчислення значення y за іншою формулою
  }

  cout << "Y = " << y << endl;  // Виведення значення y на екран

  return 0;  // Завершення програми і повернення значення 0, що означає успішне завершення
}
